;  Logic and shift instructions

      lea     R1,$00ff[R0]
      lea     R2,$0f0f[R0]

; inv R1
      add     R3,R0,R0
      add     R4,R0,R0
      add     R5,R0,R0
      add     R6,R0,R0
      logicb  R3,R1,R2,12,0   ; 8000
      logicb  R4,R1,R2,12,4   ; 0800
      logicb  R5,R1,R2,12,8   ;
      logicb  R6,R1,R2,12,12

      lea     R3,$ffff[R0]
      lea     R4,$ffff[R0]
      lea     R5,$ffff[R0]
      lea     R6,$ffff[R0]
      logicb  R3,R1,R2,12,0   ;
      logicb  R4,R1,R2,12,4   ;
      logicb  R5,R1,R2,12,8   ;
      logicb  R6,R1,R2,12,12

; and
      add     R3,R0,R0
      add     R4,R0,R0
      add     R5,R0,R0
      add     R6,R0,R0
      logicb  R3,R1,R2,1,0
      logicb  R4,R1,R2,1,4
      logicb  R5,R1,R2,1,8
      logicb  R6,R1,R2,1,12

      lea     R3,$ffff[R0]
      lea     R4,$ffff[R0]
      lea     R5,$ffff[R0]
      lea     R6,$ffff[R0]
      logicb  R3,R1,R2,1,0
      logicb  R4,R1,R2,1,4
      logicb  R5,R1,R2,1,8
      logicb  R6,R1,R2,1,12

      logicb  R3,R1,R2,7,0

      logicb  R3,R1,R2,6,0
      logicb  R4,R1,R2,6,4
      logicb  R5,R1,R2,6,8
      logicb  R6,R1,R2,6,12

      lea     R3,32[R0]     ; 0020
      shiftr  R4,R3,1       ; expect  16 = 0010
      shiftr  R5,R3,3       ; expect   4 = 0004
      shiftl  R6,R3,1       ; expect  64 = 0040
      shiftl  R7,R3,3       ; expect 256 = 0100


; Ordinary case: adding two small integers, one negative

      load   R1,k1[R0]     ; R1 = k1 = -73
      load   R2,k2[R0]     ; R2 = k2 = 250
      add    R3,R1,R2      ; -73 + 250 = 177    = $00b1


      load   R1,all1s[R0]  ; R1 = -1 = $ffff
      lea    R2,1[R0]      ; R2 =  1 = $0001
      add    R3,R1,R2      ; R3 = $0000, R15 = $0000

      trap   R0,R0,R0      ; terminate

k1    data   -73
k2    data   250
all1s data   -1    = $ffff
