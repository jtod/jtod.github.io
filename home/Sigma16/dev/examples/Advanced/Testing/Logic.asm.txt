;  Logic and shift instructions

; Shifting

      lea     R1,$0040[R0]  ; 0000 0000 0100 0000
      shiftr  R2,R1,1       ; 0000 0000 0100 0000 -> 0000 0000 0010 0000 = 0020
      shiftr  R2,R2,1       ; 0000 0000 0010 0000 -> 0000 0000 0001 0000 = 0010
      shiftr  R2,R2,1       ; 0000 0000 0001 0000 -> 0000 0000 0000 1000 = 0008
      shiftr  R3,R1,5       ; 0000 0000 0100 0000 -> 0000 0000 0000 0010 = 0002

      shiftl  R4,R1,1       ; 0000 0000 0100 0000 -> 0000 0000 1000 0000 = 0080
      shiftl  R5,R1,7       ; 0000 0000 0100 0000 -> 0010 0000 0000 0000 = 2000

; Logic
      lea     R1,$00ff[R0]
      lea     R2,$0f0f[R0]

; inv R1
      add     R3,R0,R0
      add     R4,R0,R0
      add     R5,R0,R0
      add     R6,R0,R0
      logicb  R3,R1,R2,12,0   ; 8000
      logicb  R4,R1,R2,12,4   ; 0800
      logicb  R5,R1,R2,12,8   ;
      logicb  R6,R1,R2,12,12

      lea     R3,$ffff[R0]
      lea     R4,$ffff[R0]
      lea     R5,$ffff[R0]
      lea     R6,$ffff[R0]
      logicb  R3,R1,R2,12,0   ;
      logicb  R4,R1,R2,12,4   ;
      logicb  R5,R1,R2,12,8   ;
      logicb  R6,R1,R2,12,12

; and
      add     R3,R0,R0
      add     R4,R0,R0
      add     R5,R0,R0
      add     R6,R0,R0
      logicb  R3,R1,R2,1,0
      logicb  R4,R1,R2,1,4
      logicb  R5,R1,R2,1,8
      logicb  R6,R1,R2,1,12

      lea     R3,$ffff[R0]
      lea     R4,$ffff[R0]
      lea     R5,$ffff[R0]
      lea     R6,$ffff[R0]
      logicb  R3,R1,R2,1,0
      logicb  R4,R1,R2,1,4
      logicb  R5,R1,R2,1,8
      logicb  R6,R1,R2,1,12

; or
      logicb  R3,R1,R2,7,0
      logicb  R4,R1,R2,7,4
      logicb  R5,R1,R2,7,8
      logicb  R6,R1,R2,7,12
; xor
      logicb  R3,R1,R2,6,0
      logicb  R4,R1,R2,6,4
      logicb  R5,R1,R2,6,8
      logicb  R6,R1,R2,6,12

; Ordinary case: adding two small integers, one negative

      load   R1,k1[R0]     ; R1 = k1 = -73
      load   R2,k2[R0]     ; R2 = k2 = 250
      add    R3,R1,R2      ; -73 + 250 = 177    = $00b1


      load   R1,all1s[R0]  ; R1 = -1 = $ffff
      lea    R2,1[R0]      ; R2 =  1 = $0001
      add    R3,R1,R2      ; R3 = $0000, R15 = $0000

      trap   R0,R0,R0      ; terminate

k1    data   -73
k2    data   250
all1s data   -1    = $ffff
